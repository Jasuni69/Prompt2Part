  [ [ 0, 9, 4],[ 4, 9, 8],[ 8, 9, 5],[ 5, 9, 1],[ 1, 9, 0],
    [ 2,11, 3],[ 3,11, 7],[ 7,11,10],[10,11, 6],[ 6,11, 2],
