function _xu_turn(turtle, a) = 
    let(
        unit_vts = _unit_vts(turtle),
        xu = unit_vts.x,
        m = _q_rotation(a, xu)
    )
    _create(
        _pt(turtle),
        [
            xu, 
            m * unit_vts.y, 
            m * unit_vts.z
        ]
    );

// turn the turtle around the y'-axis
// return a new unit vector
