function patch_rotate(patch, a=undef, v=undef, cp=[0,0,0]) =
	v==undef?
		[for(row=patch) rotate_points3d(row, a, cp)] :
		[for(row=patch) rotate_points3d_around_axis(row, a, v, cp)];
