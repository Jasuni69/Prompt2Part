module enviro_assembly() hflip(exploded())
assembly("enviro") {
    if(show_enviroplus)
        pcb(pcb);

    pcb_grid(pcb, 5.5, 2.5, -pcb_thickness(pcb)) {
        hflip()
            explode(15)
                fan_controller_assembly();
    }
    if(!exploded())
        for(x = [8, 7, 3])
            pcb_grid(pcb, x, 0, 0.05)
                solder_meniscus(0.39, 1);
}
