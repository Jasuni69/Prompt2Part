module test_is_lower() {
    for (i=[32:126]) {
        if (
            (i>=ord("a") && i <=ord("z"))
        ) {
            assert(is_lower(chr(i)));
        } else {
            assert(!is_lower(chr(i)));
        }
    }
    assert(is_lower("abcdefghijklmnopqrstuvwxyz"));
    assert(!is_lower("ABCDEFGHIJKLMNOPQRSTUVWXYZ"));
    assert(!is_lower("abcdefghijKlmnopqrstuvwxyz"));
    assert(!is_lower("abcdefghijklmnopqrstuvwxyZ"));
}
test_is_lower();
