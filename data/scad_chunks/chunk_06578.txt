module test_is_struct() {
    assert(is_struct([["Foo",1],["Bar",2],["Baz",3]]));
    assert(!is_struct([["Foo"],["Bar"],["Baz"]]));
    assert(!is_struct(["Foo","Bar","Baz"]));
    assert(!is_struct([3,4,5]));
    assert(!is_struct(3));
    assert(!is_struct(true));
    assert(!is_struct("foo"));
    assert(is_struct([]));
}
test_is_struct();
