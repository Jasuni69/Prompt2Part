function _nvl(v, l, s, t, degree, knots, d, i) = 
    i == (s - degree - 1 + l) ? v :
    let(
        leng_v = len(v),
        nv = [
            each [for(j = 0; j < i; j = j + 1) v[j]],
            _bspline_curve_nvi(v, i, l, t, degree, knots, d),
            each [for(j = i + 1; j < leng_v; j = j + 1) v[j]]
        ]
    )
    _nvl(nv, l, s, t, degree, knots, d, i - 1);
