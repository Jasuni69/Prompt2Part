function equiChamfer(botTop,side,fnCorner=7,sideDiff=($skinThick*2)) =

			?chamfer( botTop, botTop, side-sideDiff,fnCorner = fnCorner)
			:chamfer( botTop, botTop, side         ,fnCorner = fnCorner);
