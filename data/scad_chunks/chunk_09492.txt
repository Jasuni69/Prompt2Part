module deprecate_argument(name, arg, suggest=undef) {
	echo_warning(pfx="DEPRECATED ARG", str(
		"In `<code>", name, "</code>`, ",
		"the argument `<code>", arg, "</code>` ",
		"is deprecated and should not be used.",
		!is_def(suggest)? "" : str(
			"  You should use `<code>", suggest, "</code>` instead."
		)
	));
}
