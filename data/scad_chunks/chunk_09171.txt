function vx_curve(points, tightness = 0) = 
    let(leng = len(points))
    dedup([
        each [
            for(i = [0:leng - 4])
                let(
                    pts = _vx_catmull_rom_spline_4pts([for(j = [i:i + 3]) points[j]], tightness)
                )
                for(i = [0:len(pts) - 2]) pts[i]    
        ],
        points[leng - 2]
    ], hash = len(points[0]) == 2 ? _pt2_hash : _pt3_hash);
