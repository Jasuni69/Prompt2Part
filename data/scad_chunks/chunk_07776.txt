module melted_clock() {

    $fn = 48;
    sa = r * angle / (r + thickness);
    rotate([0, 180, 0]) 
    mirror([1, 0, 0]) {
        color("Gold") 
        translate([0, 0, thickness]) 
        melt(r, thickness, angle, frags, xscale) 
            circle(r);
        
        color("Gainsboro") 
        melt(r, thickness, sa, frags, xscale) 
            clock(r - thickness * 2, thickness, hour, minute);
    }
}

melted_clock();
