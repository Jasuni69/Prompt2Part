module drag_chain_assembly(type, pos = 0, render = false)  //! Drag chain assembly
    assembly(str(drag_chain_name(type), "_drag_chain"), big = true, ngb = true)
        if($children == 2)
            _drag_chain_assembly(type, pos, render) {
                children(0);
                children(1);
            }
        else if($children == 1)
            _drag_chain_assembly(type, pos, render)
                children(0);
        else
            _drag_chain_assembly(type, pos, render);
