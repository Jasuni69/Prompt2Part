module hex2(d1, k, s, b1, b2, b3, l){
	b = (l < 125) ? b1 :
		(l < 200) ? b2 :
		b3;
	BOLTS_check_dimension_defined(b, "threaded shaft length b");

	union(){
		BOLTS_hex_head(k,s);
		//unthreaded shaft
		cylinder(r=d1/2,h=l-b);
		//threaded shaft
		translate([0,0,l-b]) BOLTS_thread_external(d1,b);
	}
}
