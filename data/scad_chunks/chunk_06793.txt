module test_circle() {
    for (pt = circle(d=200)) {
        assert(approx(norm(pt),100));
    }
    for (pt = circle(r=100)) {
        assert(approx(norm(pt),100));
    }
    assert(is_polygon_clockwise(circle(d=200)));
    assert(is_polygon_clockwise(circle(r=100)));
    assert(len(circle(d=100,$fn=6)) == 6);
    assert(len(circle(d=100,$fn=36)) == 36);
}
test_circle();
