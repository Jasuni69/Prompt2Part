function triInkreis(points=[[0,0],[0,10],[10,0]])=
let(
a=points[0],b=points[1],c=points[2], // points abc
sA=norm(c-b), sB=norm(c-a), sC=norm(b-a), //side length
u=sA+sB+sC// circumference
)
[ (sA*a.x+sB*b.x+sC*c.x)/u ,  (sA*a.y+sB*b.y+sC*c.y)/u ];
