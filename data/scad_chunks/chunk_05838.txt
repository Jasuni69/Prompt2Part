function unit(v, error=[[["ASSERT"]]]) =
    assert(is_vector(v), "\nInvalid vector.")
    norm(v)<EPSILON? (error==[[["ASSERT"]]]? assert(norm(v)>=EPSILON,"\nCannot normalize a zero vector.") : error) :
    v/norm(v);
