function root_radius(mm_per_tooth=5, number_of_teeth=11, clearance=undef, interior=false)
	= pitch_radius(mm_per_tooth, number_of_teeth) -
	(interior? adendum(mm_per_tooth) : dedendum(mm_per_tooth, clearance));
