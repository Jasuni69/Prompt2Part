function vx_sphere(radius, filled = false, thickness = 1) = 
    let(range = [-radius: radius - 1])
    filled ? [
        for(z = range, y = range, x = range)
        let(v = [x, y, z])
        if(v * v < radius ^ 2) v
    ] :
    let(ishell = radius * radius - 2 * thickness * radius)
    [
        for(z = range, y = range, x = range)
        let(
            v = [x, y, z],
            vv = v * v
        )
        if(vv < radius ^ 2 && vv > ishell) v    
    ];
