function zip(lts, combine) = 
    let(end_lts = len(lts) - 1)
    is_undef(combine) ? 
        [
            for(i = [0:len(lts[0]) - 1])
            [for(j = [0:end_lts]) lts[j][i]]
        ] :
        [
            for(i = [0:len(lts[0]) - 1])
            combine([for(j = [0:end_lts]) lts[j][i]])
        ];
