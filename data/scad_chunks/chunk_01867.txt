module bulldog_shape(depth, height, radius, tube, open) {
        rounded_polygon([
            [-depth / 2 + radius, -height / 2 + radius, radius],
            [-depth / 2 + radius,  height / 2 - radius, radius],
            [ depth / 2 - tube,    open / 2 + tube,    -tube],
            [ depth / 2 + tube,    height / 2,          tube],
            [ depth / 2 + tube,   -height / 2,          tube],
            [ depth / 2 - tube,   -open / 2 - tube,    -tube],

        ]);
}
