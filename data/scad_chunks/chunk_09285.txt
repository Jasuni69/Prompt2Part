function sampling_noAdjacentNeighbor(s, sample, x, y) = 
    let(me = [x, y])
    every(
        NBRS,
        function(NBR)
            let(nbr = me + NBR)
            !sampling_inRow(s, nbr.y) ||
                          (let(p = sampling_grid(s)[nbr.y][nbr.x])
                          is_undef(p) ||
                          norm(sample - p) >= sampling_r(s))
    );
