module show_path(path, r = 0.1) //! Show a path using a chain of hulls for debugging, duplicate points are highlighted.
    for(i = [0 : len(path) - 2]) {
        hull($fn = 16) {
            translate(path[i])
                sphere(r);

            translate(path[i + 1])
                sphere(r);
        }
        if(path[i] == path[i + 1])
            translate(path[i])
                color("red") sphere($fn = 16, r * 4);
    }
