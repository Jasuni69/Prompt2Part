module do_list(list, show_metal) {
    layout([for (p = list) max_d]) let(p = list[$i]) {
        rotate(-145)
            if($preview)
                printed_pulley_assembly(p);
            else
                printed_pulley(p);

        if(show_metal)
            not_on_bom()
                translate([0, 60])
                    rotate(-145)
                        pulley_assembly(p);
    }
}
