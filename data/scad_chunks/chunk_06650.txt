module test_is_hexdigit() {
    for (i=[32:126]) {
        if (
            (i>=ord("0") && i <=ord("9")) ||
            (i>=ord("A") && i <=ord("F")) ||
            (i>=ord("a") && i <=ord("f"))
        ) {
            assert(is_hexdigit(chr(i)));
        } else {
            assert(!is_hexdigit(chr(i)));
        }
    }
}
test_is_hexdigit();
