function arc(r=10,deg=90,r2,rot=0,t=[0,0,0],z,fn=36,rev=false)=
  let(
    step=deg/fn
  )
  [for (i=rev?[fn:-1:0]:[0:fn])
    let(
      r=is_undef(r2)?r:r+(r2-r)/fn*i
    )
    is_undef(z)?[cos(i*step+rot)*r,sin(i*step+rot)*r]+t.xy:
                [cos(i*step+rot)*r,sin(i*step+rot)*r,z]+t
  ];
