function _Quat(a,s,w) = [a[0]*s, a[1]*s, a[2]*s, w];

// Usage:
//   Quat(ax, ang);
// Description: Create a new Quaternion from axis and angle of rotation.

//   ang = Number of degrees to rotate around the axis counter-clockwise, when facing the origin.
