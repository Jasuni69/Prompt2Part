function _vx_circle(radius, filled, thickness) = 
    let(range = [-radius: radius - 1], powr = radius ^ 2)
    filled ? [
        for(y = range, x = range)    
        let(v = [x, y])
        if(v * v < powr) v
    ] :
    let(ishell = powr - 2 * thickness * radius)
    [
        for(y = range, x = range)        
        let(
            v = [x, y],
            pow_leng = v * v
        )
        if(pow_leng < powr && pow_leng > ishell) v    
    ];
