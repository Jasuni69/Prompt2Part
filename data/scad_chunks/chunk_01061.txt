module screw_keyhole(type, h = 0) { //! Make keyhole slot to accept and retain screw head
    r = screw_head_radius(type);
    extrude_if(h) {
        translate([0, - 2 * r])
            drill(r + 0.5, 0);

        hull()
            for(y = [0, -2 * r])
                translate([0, y])
                    drill(screw_clearance_radius(type), 0);
    }
}
