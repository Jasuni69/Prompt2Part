module ball_bearings() {
    small_bearings = [for(b = ball_bearings) if(bb_diameter(b) <= 13) b];
    big_bearings = [for(b = ball_bearings) if(!in(small_bearings, b)) b];

    translate([0, 0])
        do_bearings(big_bearings);

    translate([0, -25])
        do_bearings(small_bearings);
}

if($preview)
    ball_bearings();
