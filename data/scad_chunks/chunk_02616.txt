module sector(r, start_angle, end_angle) { //! Create specified sector given radius `r`, `start_angle` and `end_angle`
    R = r * sqrt(2) + 1;

    if(end_angle > start_angle)
        intersection() {
            circle4n(r);

            // A 4 triangle fan
            polygon([[0, 0],
                     for(i = [0 : 4], a = start_angle + i * (end_angle - start_angle) / 4) R * [cos(a), sin(a)] ]);
        }
}
