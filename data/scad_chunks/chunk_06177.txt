function affine3d_skew_yz(ya=0, za=0) =
    assert(is_finite(ya))
    assert(is_finite(za))
    [
        [1,       0,       0, 0],
        [0,       1, tan(ya), 0],
        [0, tan(za),       1, 0],
        [0,       0,       0, 1]
    ];
