module chamfer_hole(depth = 1) {    //! Chamfer a hole given a 2D outline as a child
    for(i = [0 : $children - 1])
        hull() {
            linear_extrude(eps)
                offset(depth)
                    children(i);

            translate_z(-depth)
                linear_extrude(eps)
                    children(i);
        }
}
