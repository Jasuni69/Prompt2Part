function ShuffleList(list) =
  let(n = len(list))
  n > 0 ?
    let(index = floor(rands(0,n-0.001,1)[0]))
    let(left = index > 0 ? [for(i=[0:index-1]) list[i]] : [])
    let(right = index < n-1 ? [for(i=[index+1:n-1]) list[i]] : [])
    concat(list[index],ShuffleList(concat(left,right))) : [];
