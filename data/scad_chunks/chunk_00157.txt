module do_bearings(list) {
    diameters = [for(b = list) bb_diameter(b)];
    max = max(diameters);
    layout(diameters) let(b = list[$i])
        //translate([0, (max - bb_diameter(b)) / 2])
            ball_bearing(list[$i])
                if (bb_width(list[$i]) >= 5)
                    bearing_ball(3);
}
