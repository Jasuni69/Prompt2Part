module holes() {
    let($b = blocks[0])
        translate([0, box_mm.y / 2 - gap - $b.y / 2])
            children();

    let($b = blocks[1])
        translate([0, box_mm.y / 2 - gap - blocks[0].y - gap - $b.y / 2])
            children();

    for(side = [-1, 1])
        let($b = blocks[2])
            translate([side * (blocks[2].x + 2 * gap) / 2,
                       box_mm.y / 2 - gap - blocks[0].y - gap - blocks[1].y - gap - blocks[2].y / 2 ])
                children();
}
