module test_is_str() {
    assert(!is_str(undef));
    assert(!is_str(true));
    assert(!is_str(false));
    assert(!is_str(-123));
    assert(!is_str(0));
    assert(!is_str(123));
    assert(is_str(""));
    assert(is_str("foo"));
    assert(!is_str([]));
    assert(!is_str([3,4,5]));
    assert(!is_str(["foo","bar","baz"]));
    assert(!is_str([0:5]));
}
test_is_str();
