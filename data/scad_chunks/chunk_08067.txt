    let(
        axiom = "A",
        rules = [
            ["A", "B-F+CFC+F-D&F^D-F+&&CFC+F+B//"],
            ["B", "A&F^CFB^F^D^^-F-D^|F^B|FC^F^A//"],

            ["D", "|CFB-F+B|FA&F^A&&FB-F+B|FC//"]
        ]
    )
