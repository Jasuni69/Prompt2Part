function init(x, y, ix, iy, init_u, init_v) = 
    let(
	    half_x = floor(x / 2),
		half_y = floor(y / 2),
		half_ix = floor(ix / 2),
		half_iy = floor(iy / 2),
		lowx = half_x - half_ix,
		highx = half_x + half_ix,
		lowy = half_y - half_iy,
		highy = half_y + half_iy
	)
	[
		for(j = [0:y - 1])
		[
			for(i = [0:x - 1])
			i >= lowx && i < highx && j >= lowy && j < highy ?
			[init_u, init_v] : [1, 0]
		]
	];
