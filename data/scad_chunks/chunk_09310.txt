    for(n = [0:sampling_k(s) - 1])
    let(

        w = sampling_w(s),
        x = floor(sample.x / w),
        y = floor(sample.y / w),
        z = floor(sample.z / w)
    )
    if(sampling_inGrid(s, x, y, z) && sampling_noAdjacentNeighbor(s, sample, x, y, z))
    sample
];
