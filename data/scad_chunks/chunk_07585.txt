module PrevPos(on=true,t=[0,0,0],rot=[180,0,0],help){
if($preview&&on||on==2)translate(v3(t))rotate(v3(rot))children();
else children();

Echo("Render with PrevPos!",color="warning",condition=on==2);

MO(!$children);

HelpTxt("PrevPos",["on",on,"t",t,"rot",rot],help);
};
