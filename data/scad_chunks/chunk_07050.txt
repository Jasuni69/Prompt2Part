module test_path_torsion(){
    c = path3d(circle(r=1, $fn=100));
    tc = path_torsion(c, closed=true);
    assert(all_zero(tc));
    a=3;b=7;
    helix = [for(t=[0:1:20]) [a*cos(t), a*sin(t), b*t*PI/180]];
    th = path_torsion(helix, closed=false);
    assert(approx(th[5], b/(a*a+b*b), 1e-5));
}
test_path_torsion();
