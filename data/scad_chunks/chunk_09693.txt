module torx_drive(size, l=5, center=undef, orient=ORIENT_Z, align=V_UP) {
	od = torx_outer_diam(size);
	orient_and_align([od, od, l], orient, align, center) {
		linear_extrude(height=l, convexity=4, center=true) {
			torx_drive2d(size);
		}
	}
}
