module RPI_assembly()
assembly("RPI") {
    pcb(rpi);

    pcb_grid(rpi, 9.5, 0.5, -0.6)
        rotate(180)
            explode(20)
                pin_header(2p54header, 20, 2, right_angle = true);
}
