module test_is_nan() {
    assert(!is_nan(undef));
    assert(!is_nan(true));
    assert(!is_nan(false));
    assert(!is_nan(-5));
    assert(!is_nan(0));
    assert(!is_nan(5));
    assert(!is_nan(INF));
    assert(!is_nan(-INF));
    assert(!is_nan(""));
    assert(!is_nan("foo"));
    assert(!is_nan([]));
    assert(!is_nan([3,4,5]));
    assert(!is_nan([3:1:5]));
    assert(is_nan(NAN));
}
test_is_nan();
