function sorted(lt, cmp = undef, key = undef, reverse = false) = 
    let(is_cmp_undef = is_undef(cmp))
    is_cmp_undef && is_undef(key) ? _sorted_default(lt, reverse) :
    is_cmp_undef ?                  _sorted_key(lt, key, reverse) :
                                    _sorted_cmp(lt, cmp, reverse);
