module rounded_polygons() {
    tangents = rounded_polygon_tangents(profile);
    length = rounded_polygon_length(profile, tangents);

    rotate([70, 0, 315])
        linear_extrude(eps)
            rounded_polygon(profile, tangents);

    translate([0, -10])
        label(str("perimeter length = ", length), valign = "top", halign = "right");

}

rounded_polygons();
