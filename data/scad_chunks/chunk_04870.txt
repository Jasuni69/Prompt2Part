module fillet(r, s){
    data = _pb_fillet($pb_pts, [r, s], $pb_angle);
    $pb_post = concat($pb_post, data[1]);  //  fillet tag
    $pb_angle = data[2];
    $pb_ctrl_pts = data[3];
    if (pb_do_render($children, parent_module(0))) pb_draw();
    children();  
}
