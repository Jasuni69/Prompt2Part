module test_constrain() {
	assert(constrain(-2,-1,1) == -1);
	assert(constrain(-1.75,-1,1) == -1);
	assert(constrain(-1,-1,1) == -1);
	assert(constrain(-0.75,-1,1) == -0.75);
	assert(constrain(0,-1,1) == 0);
	assert(constrain(0.75,-1,1) == 0.75);
	assert(constrain(1,-1,1) == 1);
	assert(constrain(1.75,-1,1) == 1);
	assert(constrain(2,-1,1) == 1);
}
test_constrain();
