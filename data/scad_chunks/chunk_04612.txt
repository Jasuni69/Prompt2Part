module decorate_front( d ) { 
    yd = d.y*0.7; yofs=0;
    xd = d.x-(corner_radius*2);
    //translate( [(d.x/2) - (xd/2), yofs+ ((d.y/2) - (yd/2)), -0.1] )
    translate( [(d.x/2) - (xd/2), yofs+ ((d.y/2) - (yd/2)), -0.1] )
    translate([0,yd,0]) rotate([0,0,-90])
    resize( [yd, xd, wall_thick] ) {
        linear_extrude( height=1) { text( nametext, font="Sans"); } 
    }
}
