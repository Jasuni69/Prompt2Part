function modular_hose_radius(size, outer=false) =
  let(
      ind = search([size],[1/4, 1/2, 3/4])[0]
  )
  assert(ind!=[], "Must specify size as 1/4, 1/2 or 3/4")
  let(
     b = select(_modhose_big_end[ind], [0,-1]),
     s = select(_modhose_small_end[ind], [0,-1])
  )
  outer ? b[1][0] : b[0][0];
