module test_f_is_undef() {
    testfn = f_is_undef();
    for (test = [
        [undef,   true],
        [false,   false],
        [true,    false],
        [0,       false],
        [3,       false],
        ["foo",   false],
        [[4,5,6], false]
    ]) {
        assert(testfn(test[0]) == test[1]);
    }
}
test_f_is_undef();
