module do_corner_block(screw)
    if($preview) {
        fastened_corner_block_assembly(3, screw = screw);

        translate([0, 30])
            fastened_2screw_block_assembly(3, screw = screw);
    }
    else {
        corner_block(screw);

        translate([0, 30])
            2screw_block(screw);
    }
