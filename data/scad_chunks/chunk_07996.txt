function fern(n = 8, angle = 4, leng = 1, heading = 0, start = [0, 0]) = 
    let(
        axiom = "EEEA",
        rules = [
            ["A", "[++++++++++++++EC]B+B[--------------ED]B+BA"],
            ["C", "[---------EE][+++++++++EE]B+C"],
            ["D", "[---------EE][+++++++++EE]B-D"]
        ]
    )
    lsystem2(axiom, rules, n, angle, leng, heading, start, forward_chars = "ABCDE");
