module yslots (d, scl, hsz, t=wall_thick ) {
    // vent slots constrained within scaled space
    sd = d*scl;
    iv = hsz;
    xcx = sd.x / iv; ycx = sd.y / iv;
    xrt = (d.x*(1.0-scl)/2); yrt = (d.y*(1.0-scl)/2);
    for (iy=[1:ycx-1]) {
        ypt = iy * iv;
        translate( [xrt,yrt+ypt,0] ) cube( [sd.x, t, d.z] );
    }
}
