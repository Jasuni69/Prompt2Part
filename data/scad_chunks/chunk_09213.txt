function _ydominant(start, end, a, s) = 
    let(   
        shry = floor(a.y / 2),
        xd = a.x - shry,
        zd = a.z - shry
    )
    [
        start,
        each _ydominant_sub(
            _ydominant_x(start.x, xd, s.x), 
            start.y + s.y,
            _ydominant_z(start.z, zd, s.z), 
            end.y, 
            a, 
            s, 
            _ydominant_xd(xd, a.x, a.y), 
            _ydominant_zd(zd, a.y, a.z)
        )
    ];
