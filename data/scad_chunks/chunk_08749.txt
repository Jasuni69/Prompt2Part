function hashmap_del(map, key, eq = undef, hash = _str_hash) =
    let(
	    bidx = hash(key) % len(map),
		bucket = map[bidx],
		leng_bucket = len(bucket),
		leng_map = len(map)
	)
	leng_bucket == 0 ? map :
	let(i = _find_eq(bucket, key, eq))
	i == -1 ? map : 
	[
	    for(j = 0; j < leng_map; j = j + 1) j == bidx ? 
	        [for(k = 0; k < leng_bucket; k = k + 1) if(k != i) bucket[k]] :
			map[j]
	];
