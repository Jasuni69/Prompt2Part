module test_f_is_int() {
    testfn = f_is_int();
    for (test = [
        [undef,   false],
        [false,   false],
        [true,    false],
        [-4,      true],
        [-4.5,    false],
        [0,       true],
        [1.5,     false],
        [3,       true],
        [INF,     false],
        [-INF,    false],
        [NAN,     false],
        ["foo",   false],
        [[4,5,6], false]
    ]) {
        assert(testfn(test[0]) == test[1]);
    }
}
test_f_is_int();
