module Demo(n=0) {
  demo_cnt = 5;
  demo_cnt_sqrt = floor(sqrt(demo_cnt));
  demo_order = [2, 3, 4, 1, 5];
  rotate_demos = 270;
  //rotate_demos = 180;
  if (n == 0) {
    for (i=[1:demo_cnt])
      translate([((i-1)%demo_cnt_sqrt)*60, floor((i-1)/demo_cnt_sqrt)*60, 0])
        rotate([0, 0, rotate_demos])
        if (i-1 < len(demo_order)) {
          DemoNumber(demo_order[i-1]);
        }
        else{
          DemoNumber(i);
        }
  }
  else {
    DemoNumber(n);
  }
}
