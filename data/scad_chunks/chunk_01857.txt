module sbr_screw_locations(type, l) { //! Linear locations of screws
    P = type[6];

    count = floor(l / P);
    first = (l - count * P)/2;
    N = first == 0 ? P/2 : first;  // we don't want screws right on the edge

    for (x = [N:P:l])
        translate([0,0,l/2 - x])
            children();
}
