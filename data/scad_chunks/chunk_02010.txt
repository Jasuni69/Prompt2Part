module screw_lug(screw, h = 0) //! Create a D shaped lug for a screw
    extrude_if(h, center = false)
        difference() {
            r = screw_lug_radius(screw);
            hull() {
                circle4n(r);

                translate([-r, -r])
                    square([2 * r, eps]);
            }
            poly_circle(screw_clearance_radius(screw));
        }
