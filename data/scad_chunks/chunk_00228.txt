module displays()
    layout([for(d = displays) pcb_length(display_pcb(d))], 10)
        let(d = displays[$i], offset = display_pcb_offset(d))
            translate([-offset.x, pcb_width(display_pcb(d)) / 2 - offset.y])
                vflip()
                    display(d);

if($preview)
    let($show_threads = true)
        displays();
