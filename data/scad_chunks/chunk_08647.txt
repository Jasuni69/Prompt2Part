function _yu_turn(turtle, a) = 
    let(
        unit_vts = _unit_vts(turtle),
        yu = unit_vts.y,
        m = _q_rotation(a, yu)
    )
    _create(
        _pt(turtle),
        [
            m * unit_vts.x, 
            yu, 
            m * unit_vts.z
        ]
    );

// turn the turtle around the z'-axis
// return a new unit vector
