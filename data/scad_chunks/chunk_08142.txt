module rock_wall(p1, p2, size) {
    dvt = p2 - p1;
    leng = norm(dvt);
	uvt = dvt / leng;
	
    along_with([for(i = [0:3:leng]) p1 + uvt * i], method = "EULER_ANGLE")
	    rock(size);
		
	translate([0, 0, 1.2])
    along_with([for(i = [0:4:leng - 1]) p1 + uvt * i], method = "EULER_ANGLE")
	    rock(size * 0.875);
}
