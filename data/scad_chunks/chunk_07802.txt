module big_caterpillar(radius) {
	translate([0, -radius * 4, 0]) 
	rotate([90, 0, 0]) 
	    track(radius);
	
	translate([0, 0, radius * 3]) 
	    body(radius);
	
	translate([0, radius * 4, 0]) 
	rotate([90, 0, 0]) 
	    track(radius);
	
	color("yellow") 
	translate([radius * 6, -radius * 4.5, radius * 9.5]) 
	rotate([90, 135, 0]) 
	    arm(radius);
	
	translate([radius * 10.75, -radius * 4.5, radius / 2.325]) 
	rotate([0, 70, 180]) 
	    glove(radius);
}
