module test_quantdn() {
	assert(quantdn(-4,3) == -6);
	assert(quantdn(-3,3) == -3);
	assert(quantdn(-2,3) == -3);
	assert(quantdn(-1,3) == -3);
	assert(quantdn(0,3) == 0);
	assert(quantdn(1,3) == 0);
	assert(quantdn(2,3) == 0);
	assert(quantdn(3,3) == 3);
	assert(quantdn(4,3) == 3);
	assert(quantdn(7,3) == 6);
}
test_quantdn();
