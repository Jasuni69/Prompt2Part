function Hue(hue) =
  let (h = (hue/60)%6)  // change to 0...6
  h < 1 ? [1,h,0] :     // 0...1
  h < 2 ? [2-h,1,0] :   // 1...2
  h < 3 ? [0,1,h-2] :   // 2...3
  h < 4 ? [0,4-h,1] :   // 3...4
  h < 5 ? [h-4,0,1] :   // 4...5
  [1, 0, 6-h];          // 5...6
