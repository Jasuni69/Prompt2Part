function vx_bezier(p1, p2, p3, p4) = 
    let(
        is2d = len(p1) == 2,
        pts = is2d ? _vx_bezier2(__to3d(p1), __to3d(p2), __to3d(p3), __to3d(p4), []) :
                     _vx_bezier3(p1, p2, p3, p4, []),
        deduped = dedup(pts, hash = is2d ? _pt2_hash : _pt3_hash)
    )
    is2d ? [for(p = deduped) __to2d(p)] : deduped;
