module aluProExtrusionBetweenPoints(p1,p2,profileType=BR_20x20,roll=0) {
	v = subv(p2,p1);
	l = mod(v);
	translate(p1) orientate(v,roll=roll) aluProExtrusion(profileType, l);
}
