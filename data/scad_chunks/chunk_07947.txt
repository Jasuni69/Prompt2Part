module petals(fibseq, i = 0) {
    if(i < len(fibseq) - 1) {
        f1 = fibseq[i];
        f2 = fibseq[i + 1];
        offset = f1 - f2;
        translate([0, 0, offset * 1.8]) 
        rotate([-5, 40, 10])
            petal(-offset, -offset / 6);   

        translate([0, offset, 0]) 
        rotate(90) 
            petals(fibseq, i + 1);         
    }
}
