module RPI_case_assembly() pose([ 20.50, 0.00, 153.30 ])
assembly("RPI_case") {
    enviro_case_assembly();

    translate([pcb_x, 0]) {
        translate_z(top_thickness) {

            translate(rpi_offset)
                rotate([-90, 0, 0])
                    RPI_assembly();
        }
        rpi_holes()
            vflip()
                translate_z(pcb_thickness(rpi))
                    screw(pcb_screw, 6.4);
    }
}
