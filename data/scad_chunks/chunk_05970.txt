module chamfcube(size=[1,1,1],chamfer=0.25,chamfaxes=[1,1,1],chamfcorners=false) {
    cuboid(
        size=size, chamfer=chamfer,
        trimcorners=chamfcorners,
        edges=[
            if (chamfaxes.x) "X",
            if (chamfaxes.y) "Y",
            if (chamfaxes.z) "Z",
        ]
    );
}
