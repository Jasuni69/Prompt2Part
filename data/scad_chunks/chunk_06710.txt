module test_filter() {
    l = [7,3,9,1,6,1,3,2];
    lt3 = function (x) x<3;
    lte3 = function (x) x<=3;
    gt3 = function (x) x>3;
    gte3 = function (x) x>=3;
    assert_equal(filter(lt3,l), [1,1,2]);
    assert_equal(filter(lte3,l), [3,1,1,3,2]);
    assert_equal(filter(gt3,l), [7,9,6]);
    assert_equal(filter(gte3,l), [7,3,9,6,3]);
}
test_filter();
