function _nz_worley_border(p, cells) = 
    let(
        dists = [for(cell = cells) norm(cell - p)],
        m1 = min(dists),
        i1 = search(m1, dists)[0],
        dists2 = [for(i = [0:len(dists) - 1]) if(i != i1) dists[i]],
        nbrs2 = [for(i = [0:len(cells) - 1]) if(i != i1) cells[i]],
        m2 = min(dists2),
        i2 = search(m2, dists2)[0],
        fst = cells[i1],
        snd = nbrs2[i2],
        m = (fst + snd) / 2        
    )
    (p - m) * (fst - m);
