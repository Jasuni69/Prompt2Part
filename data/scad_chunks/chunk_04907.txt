function _os_offset_poly(p, dist) = [ 
  for (i=[0:len(p)-1]) 
    i == 0 ? _os_offset_corner(p[len(p)-1], p[i], p[i+1], dist) :
    i == len(p)-1 ? _os_offset_corner(p[i-1], p[i], p[0], dist) :
    _os_offset_corner(p[i-1], p[i], p[i+1], dist)
  ];
