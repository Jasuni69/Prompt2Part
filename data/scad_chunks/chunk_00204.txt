module led_meters()
    layout([for(m = led_meters) meter_bezel_length(m)], 5) let(m = led_meters[$i])
        if($preview) {
            hflip()
                meter(m, colour = "blue", value = "123");

            if(!$i)
                translate([0, meter_bezel_width(m)])
                    meter_assembly(m, value = "123");
        }
        else
            meter_bezel(m);

led_meters();
