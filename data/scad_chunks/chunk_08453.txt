function geom_polar_zonohedra(n, theta = 35.5) = 
    let(
        points = flat([
            for(i = [0:n - 1], j = [i + 1:n + i - 1])
                rhombi(i, j, n, theta)
        ]),
        faces = [
            for(i = [0:len(points) / 4 - 1]) 
            [for(j = [0:3]) i * 4 + j]
        ]
    )
    [points, faces];
