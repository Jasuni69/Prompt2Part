module ripples() {
    point_size = size / mesh_w;

	sf = [
		for(y = [0:point_size.y - 1]) 
		[
			for(x = [0:point_size.x - 1]) 
			let(
				px = x * mesh_w,
				py = y * mesh_w,
				nz = nz_worley2(px, py, seed, grid_w, dist)[2],
				n = amplitude * nz_perlin2(nz / wave_smoothing, nz / wave_smoothing, seed)
			)
			[px, py, n]
		]
	];

	sf_thicken(sf, thickness);
}
