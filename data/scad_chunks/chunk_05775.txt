function _vnf_find_edge_faces(vnf,edge) =
  let(
      faces = vnf[1],
      goodind = [for(i=idx(faces))
                    let(result=search(edge,faces[i]))
                    if (result*0==[0,0] && 
                          (abs(result[0]-result[1])==1
                           || (min(result)==0 && max(result)==len(faces[i])-1)))
                       i
                ]
  )
  goodind;
