module diecut( d ){
    // cut children to d sized base_box shell 
    intersection(){
        translate([0,0,0]) base_box( [d.x, d.y, d.z] );
        gang(){ children( [0:1:$children-1] ); };
    }
}

    note Z dimension is NOT scaled
