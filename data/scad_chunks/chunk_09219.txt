function _zdominant(start, end, a, s) = 
    let(   
        shrz = floor(a.z / 2),
        xd = a.x - shrz,
        yd = a.y - shrz
    )
    [
        start, 
        each _zdominant_sub(
            _zdominant_x(start.x, xd, s.x), 
            _zdominant_y(start.y, yd, s.y), 
            start.z + s.z,
            end.z, 
            a, 
            s, 
            _zdominant_xd(xd, a.x, a.z), 
            _zdominant_yd(yd, a.y, a.z)
        )
    ];
