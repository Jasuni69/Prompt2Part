function _xdominant(start, end, a, s) = 
    let(     
        shrx = floor(a.x / 2),
        yd = a.y - shrx,
        zd = a.z - shrx
    )
    [
        start,
        each _xdominant_sub(
            start.x + s.x, 
            _xdominant_y(start.y, yd, s.y), 
            _xdominant_z(start.z, zd, s.z), 
            end.x, 
            a, 
            s, 
            _xdominant_yd(yd, a.x, a.y), 
            _xdominant_zd(zd, a.x, a.z)
        )
    ];
