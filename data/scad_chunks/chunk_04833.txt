module h(x){
    data = _pb_horz(pb_last($pb_pts), [x], true, $pb_angle);
    $pb_pts = concat($pb_pts, data[0]);
    $pb_angle = data[2]==undef? $pb_angle : data[2];
    $pb_ctrl_pts = data[3];
    if (pb_do_render($children, parent_module(0))) pb_draw();
    children();
}
