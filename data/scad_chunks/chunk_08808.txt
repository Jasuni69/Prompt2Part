function cell_pt(fcord, grid_w, seed, x, y, gw, gh) = 
    let(
        nx = fcord.x + x,
        ny = fcord.y + y,
        sd_x = (nx + gw) % gw,
        sd_y = (ny + gh) % gh,                 
        sd_base = abs(sd_x + sd_y * grid_w)
    )
    ([nx, ny] + rands(0.1, 0.9, 2, seed_value = seed + sd_base)) * grid_w;

// 9-nearest-neighbor
