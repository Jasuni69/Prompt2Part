module test_quantup() {
	assert(quantup(-4,3) == -3);
	assert(quantup(-3,3) == -3);
	assert(quantup(-2,3) == 0);
	assert(quantup(-1,3) == 0);
	assert(quantup(0,3) == 0);
	assert(quantup(1,3) == 3);
	assert(quantup(2,3) == 3);
	assert(quantup(3,3) == 3);
	assert(quantup(4,3) == 6);
	assert(quantup(7,3) == 9);
}
test_quantup();
