function closest_point(pt, points) =
    assert(is_vector(pt), "\nInvalid point." )
    assert(is_path(points,dim=len(pt)), "\nInvalid pointlist or incompatible dimensions." )
    min_index([for (p=points) norm(p-pt)]);
