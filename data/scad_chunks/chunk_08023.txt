function penrose_tiling(n = 2, angle = 36, leng = 1, heading = 0, start = [0, 0]) = 
    let(
        axiom = "[7]++[7]++[7]++[7]++[7]",
        rules = [
            ["6", "81++91----71[-81----61]++"],
            ["7", "+81--91[---61--71]+"],
            ["8", "-61++71[+++81++91]-"],
            ["9", "--81++++61[+91++++71]--71"]
        ]
    )
    lsystem2(axiom, rules, n, angle, leng, heading, start, "6789");
