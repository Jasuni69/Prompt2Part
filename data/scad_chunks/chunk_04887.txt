function circlePoints(r=undef, d=undef, z=undef) = let(
    _r = r!=undef? r : d!=undef? d*0.5 : 1,
    _c = _pu_segmentsPerCircle(_r),
    _s = 360 / _c
) [for(i=[0:_c-1]) let(_cos = cos(i*_s), _sin = sin(i*_s)) z==undef? [_cos * _r, _sin * _r,] : [_cos * _r, _sin * _r, z]];
