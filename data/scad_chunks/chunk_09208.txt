function _xdominant_sub(x, y, z, endx, a, s, yd, zd) = 
    x == endx ? [] : [
        [x, y, z], 
        each _xdominant_sub(
            x + s.x, 
            _xdominant_y(y, yd, s.y), 
            _xdominant_z(z, zd, s.z), 
            endx, 
            a, 
            s, 
            _xdominant_yd(yd, a.x, a.y), 
            _xdominant_zd(zd, a.x, a.z)
        )
    ];
        
// y-dominant
