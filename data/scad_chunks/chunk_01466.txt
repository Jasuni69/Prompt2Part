module carriage_hole_positions(type) { //! Position children over screw holes
    x_pitch = carriage_pitch_x(type);
    y_pitch = carriage_pitch_y(type);

    for(x = [-1, 1], y = [-1, 1])
        if(x < 0 || x_pitch)
            translate([x * x_pitch / 2, y * y_pitch / 2, carriage_height(type)])
                children();
}
