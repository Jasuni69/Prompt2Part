function TouchPoint(Center1,Radius1,Center2,Radius2) =
  let( d = Center2-Center1,
       l_total = norm(d),
       l_avg = (Radius1 + l_total-Radius2)/2,
       a = atan2(d.y,d.x))
  [Center1.x + l_avg*cos(a),
   Center1.y + l_avg*sin(a)];
