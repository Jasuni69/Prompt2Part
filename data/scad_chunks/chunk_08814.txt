function __line_intersection2(line_pts1, line_pts2, epsilon = 0.0001) = 
    let(
        a1 = line_pts1[0],
        a2 = line_pts1[1],
        b1 = line_pts2[0],
        b2 = line_pts2[1],
        a = a2 - a1, 
        b = b2 - b1,
        c = cross(a, b)
    ) 
    abs(c) < epsilon ? [] :  // they are parallel or conincident edges
    let(
         t = cross(b1 - a1, b) /  c,
         u = -cross(a1 - b1, a) / c
    )
    t >= 0 && t <= 1 && u >= 0 && u <= 1 ? a1 + a * t : [];
