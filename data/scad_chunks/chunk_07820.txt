module rock(detail) {
    radius = 16.5;
    detail = detail;
    amplitude = .25;
    dist = "border"; 
	
	difference() {
		translate([1, 20, -6.9])
		scale([1, 1.2, 1])
		rotate([35, 3, 0])
		translate([0, 0, 12])
		scale([3, 1, 1])	
			worley_sphere(radius, detail, amplitude, dist, seed = 4);

		translate([0, 34, -30]) 
		linear_extrude(30)
			square(150, center = true);
	}
}
